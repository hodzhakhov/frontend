fn read_file(path: String) -> HashSet {
  let file = File::open(path).unwrap();
  let reader = BufReader::new(file);

  let mut lines: HashSet = HashSet::new();
  
  for line in reader.lines() {
    lines.insert(line.unwrap());
  }

  lines
}
